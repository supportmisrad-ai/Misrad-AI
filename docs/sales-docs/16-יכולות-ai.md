## אבטחת AI — פירוט טכני

### סינון אוטומטי (lib/ai-security.ts)

```typescript
const SENSITIVE_FIELDS = [
  'email', 'phone', 'password',
  'hourlyRate', 'monthlySalary',
  'creditCard', 'ssn', 'idNumber',
  'bankAccount', 'socialSecurity'
];

// sanitizeForAI — מסנן שדות רגישים לפני שליחה ל-AI
function sanitizeForAI(data: any): any {
  // רקורסיבי — מסנן בכל רמת עומק
  return recursiveFilter(data, SENSITIVE_FIELDS);
}

// prepareAIContext — מתאים נתונים לפי הרשאות
function prepareAIContext(data, permissions) {
  const sanitized = sanitizeForAI(data);
  if (!permissions.includes('view_financials')) {
    delete sanitized.financialData;
  }
  return sanitized;
}

// validateAIResponse — בודק שה-AI לא הדליף
function validateAIResponse(response: string): boolean {
  const patterns = [/\b\d{9}\b/, /\b\d{4}[-\s]\d{4}\b/]; // מספרי ת.ז., כרטיסים
  return !patterns.some(p => p.test(response));
}
```
